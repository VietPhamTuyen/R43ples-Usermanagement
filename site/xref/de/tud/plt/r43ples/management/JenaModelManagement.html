3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.UnsupportedEncodingException;
8   import java.nio.charset.StandardCharsets;
9   
10  import com.hp.hpl.jena.rdf.model.Model;
11  import com.hp.hpl.jena.rdf.model.ModelFactory;
12  
13  public class JenaModelManagement {
14  	
15  	/**
16  	 * Read turtle file to jena model.
17  	 * 
18  	 * @param path the file path to the turtle file
19  	 * @return the jena model
20  	 */
21  	public static Model readTurtleFileToJenaModel(String path) {
22  		Model model = ModelFactory.createDefaultModel();
23  		InputStream is = ClassLoader.getSystemResourceAsStream(path);
24  		model.read(is, null, "TURTLE");
25  		try {
26  			is.close();
27  		} catch (IOException e) {
28  			e.printStackTrace();
29  		}
30  		
31  		return model;
32  	}
33  	
34  	
35  	
36  	/**
37  	 * Read RDF string to jena model.
38  	 * 
39  	 * @param triples the triples in RDF serialization
40  	 * @param format Jena format of triples
41  	 * @return the model
42  	 */
43  	public static Model readStringToJenaModel(String triples, String format) {
44  		Model model = ModelFactory.createDefaultModel();
45  		InputStream is = new ByteArrayInputStream(triples.getBytes(StandardCharsets.UTF_8));
46  		model.read(is, null, format);
47  		try {
48  			is.close();
49  		} catch (IOException e) {
50  			e.printStackTrace();
51  		}
52  		
53  		return model;
54  	}
55  	
56  	
57  	/**
58  	 * Read N-Triple string to jena model.
59  	 * 
60  	 * @param triples the triples in N-Triple serialization
61  	 * @return the model
62  	 */
63  	public static Model readNTripleStringToJenaModel(String triples) {
64  		return readStringToJenaModel(triples, "N-TRIPLE");
65  	}
66  	
67  	
68  	/**
69  	 * Converts a jena model to N-Triple serialization. 
70  	 * 
71  	 * @param model the jena model
72  	 * @return the string which contains the N-Triples
73  	 */
74  	public static String convertJenaModelToNTriple(Model model) {
75  			
76  		ByteArrayOutputStream os = new ByteArrayOutputStream();
77  		model.write(os, "N-TRIPLES");
78  		
79  		try {
80  			return new String(os.toByteArray(), "UTF-8");
81  		} catch (UnsupportedEncodingException e) {
82  			e.printStackTrace();
83  			return "";
84  		}
85  	}
86  
87  }
