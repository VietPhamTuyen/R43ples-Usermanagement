18

24

26  	public RevisionDraft(final String graphName, final String revisionName) throws InternalErrorException {
27  		this.graphName = graphName;
28  		this.revisionGraph = RevisionManagement.getRevisionGraph(graphName);
29  		this.revisionName = revisionName;
30  		this.revisionNumber= RevisionManagement.getRevisionNumber(revisionGraph, revisionName);
31  		
32  		this.newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName);
33  		
34  		this.addSetURI = graphName + "-addSet-" + newRevisionNumber;
35  		this.deleteSetURI = graphName + "-deleteSet-" + newRevisionNumber;
36  		this.referenceFullGraph = RevisionManagement.getReferenceGraph(graphName, revisionName);
37  	}
38  	
39  	
40  	
41  	public boolean equals(final String graphName, final String revisionNumber) throws InternalErrorException{
42  		String revisionGraph = RevisionManagement.getRevisionGraph(graphName);
43  		String revNumber = RevisionManagement.getRevisionNumber(revisionGraph, revisionNumber);
44  		return ((this.graphName.equals(graphName)) && (this.revisionNumber.equals(revNumber)));
45  	}
