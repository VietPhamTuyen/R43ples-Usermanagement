27  	 * 
28  	 * @param query R43ples query string
29  	 * @param format serialisation format of the result 
30  	 * @param join_option JOIN option
31  	 * @return string containing result of the query
32  	 * @throws InternalErrorException
33  	 */
34  	public static String sparqlSelectConstructAsk(final String query, final String format, final String join_option)
35  			throws InternalErrorException {
36  		String result;
37  		if (join_option.equals("old")) {
38  			String query_rewritten = SparqlRewriter_old.rewriteQuery(query);
39  			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
40  		} 
41  		else if (join_option.equals("new")) {
42  			String query_rewritten = SparqlRewriter.rewriteQuery(query);
43  			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
44  		}
45  		else {
46  			result = getSelectConstructAskResponseClassic(query, format);
47  		}
48  		return result;
49  	}

51  	/**
52  	 * @param query
53  	 * @param format
54  	 * @return
55  	 * @throws InternalErrorException 
56  	 */
57  	private static String getSelectConstructAskResponseClassic(final String query, final String format) throws InternalErrorException {
58  		final Pattern patternSelectFromPart = Pattern.compile(
59  				"(?<type>FROM|GRAPH)\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"",
60  				Pattern.DOTALL + Pattern.MULTILINE + Pattern.CASE_INSENSITIVE);
61  
62  		String queryM = query;
63  
64  		Matcher m = patternSelectFromPart.matcher(queryM);
65  		while (m.find()) {
66  			String graphName = m.group("graph");
67  			String type = m.group("type");
68  			String revisionNumber = m.group("revision").toLowerCase();
69  			String newGraphName;
70  
71  			// if no revision number is declared use the MASTER as default
72  			if (revisionNumber == null) {
73  				revisionNumber = "master";
74  			}
75  			if (revisionNumber.equalsIgnoreCase("master")) {
76  				// Respond with MASTER revision - nothing to be done - MASTER revisions are already created in the named graphs
77  				newGraphName = graphName;
78  			} else {
79  				if (RevisionManagement.isBranch(graphName, revisionNumber)) {
80  					newGraphName = RevisionManagement.getReferenceGraph(graphName, revisionNumber);
81  				} else {
82  					// Respond with specified revision, therefore the revision must be generated - saved in graph <RM-TEMP-graphName>
83  					newGraphName = graphName + "-temp";
84  					RevisionManagement.generateFullGraphOfRevision(graphName, revisionNumber, newGraphName);
85  				}
86  			}
87  
88  			queryM = m.replaceFirst(type + " <" + newGraphName + ">");
89  			m = patternSelectFromPart.matcher(queryM);
90  			
91  		}
92  		String response = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(queryM, format);
93  		return response;
94  	}

97  	public static void sparqlUpdate(final String query, final String user, final String commitMessage)
98  			throws InternalErrorException {
99  
100 		final Pattern patternUpdateRevision = Pattern.compile(
101 				"(?<action>INSERT|DELETE|WHERE)(?<data>\\s*DATA){0,1}\\s*\\{\\s*GRAPH\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"",
102 				patternModifier);
103 		final Pattern patternEmptyGraphPattern = Pattern.compile(
104 				"GRAPH\\s*<(?<graph>[^>]*)>\\s*\\{\\s*\\}",
105 				patternModifier);
106 		final Pattern patternGraphWithRevision = Pattern.compile(
107 				"GRAPH\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"",
108 				patternModifier);
109 		logger.info("Update detected");
110 		
111 		// write to add and delete sets
112 		// (replace graph names in query)
113 		String queryM = query;
114 		Matcher m = patternUpdateRevision.matcher(queryM);
115 		while (m.find()) {
116 			String graphName = m.group("graph");
117 			String revisionName = m.group("revision").toLowerCase(); 	// can contain revision
118 																		// numbers or reference
119 																		// names
120 			String action = m.group("action");
121 			String data = m.group("data");
122 			if (data == null)
123 				data = "";
124 			
125 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
126 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
127 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
128 			if (!RevisionManagement.isBranch(graphName, revisionName)) {
129 				throw new InternalErrorException("Revision is not referenced by a branch");
130 			}
131 			if (action.equalsIgnoreCase("INSERT")) {
132 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, addSetGraphUri));
133 			} else if (action.equalsIgnoreCase("DELETE")) {
134 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, removeSetGraphUri));
135 			} else if (action.equalsIgnoreCase("WHERE")) {
136 				// TODO: replace generateFullGraphOfRevision with SPARQL JOIN
137 				String tempGraphName = graphName + "-temp";
138 				RevisionManagement.generateFullGraphOfRevision(graphName, revisionName, tempGraphName);
139 				queryM = m.replaceFirst(String.format("WHERE { GRAPH <%s>", tempGraphName));
140 			}
141 			m = patternUpdateRevision.matcher(queryM);
142 		}
143 		
144 		// Remove empty insert clauses which otherwise will lead to errors
145 		m= patternEmptyGraphPattern.matcher(queryM);
146 		queryM = m.replaceAll("");
147 
148 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryM);
149 
150 		queryM = query;
151 		m = patternGraphWithRevision.matcher(queryM);
152 		while (m.find()) {
153 			String graphName = m.group("graph");
154 			String revisionName = m.group("revision").toLowerCase();	// can contain revision
155 																		// numbers or reference
156 																		// names
157 			// General variables
158 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
159 			String referenceFullGraph = RevisionManagement.getReferenceGraph(graphName, revisionName);
160 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
161 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
162 
163 			// remove doubled data
164 			// (already existing triples in add set; not existing triples in
165 			// delete set)
166 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
167 							"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", addSetGraphUri,
168 							referenceFullGraph));
169 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
170 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } MINUS { GRAPH <%s> { ?s ?p ?o. } } }",
171 					removeSetGraphUri, removeSetGraphUri, referenceFullGraph));
172 
173 			// merge change sets into reference graph
174 			// (copy add set to reference graph; remove delete set from reference graph)
175 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
176 						"INSERT { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }",
177 						referenceFullGraph,	addSetGraphUri));
178 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
179 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", 
180 					referenceFullGraph,	removeSetGraphUri));
181 
182 			// add meta information to R43ples
183 			ArrayList<String> usedRevisionNumber = new ArrayList<String>();
184 			usedRevisionNumber.add(revisionName);
185 			RevisionManagement.addMetaInformationForNewRevision(graphName, user, commitMessage, usedRevisionNumber,
186 					newRevisionNumber, addSetGraphUri, removeSetGraphUri);
187 			
188 			
189 			queryM = m.replaceAll(String.format("GRAPH <%s> ", graphName));
190 			m = patternGraphWithRevision.matcher(queryM);
191 		}

194 	public static String sparqlCreateGraph(final String query) throws QueryErrorException {
195 		final Pattern patternCreateGraph = Pattern.compile(
196 				"CREATE\\s*(?<silent>SILENT)?\\s*GRAPH\\s*<(?<graph>[^>]*)>",
197 				patternModifier);
198 		String graphName = null;
199 		Matcher m = patternCreateGraph.matcher(query);
200 		boolean found = false;
201 		while (m.find()) {
202 			found = true;
203 			graphName = m.group("graph");
204 //			String silent = m.group("silent");
205 			
206 			// Create graph
207 			TripleStoreInterfaceSingleton.get().executeCreateGraph(graphName);
208 		    
209 		    if (RevisionManagement.getMasterRevisionNumber(graphName) == null)
210 		    {
211 			    // Add R43ples information
212 			    RevisionManagement.putGraphUnderVersionControl(graphName);
213 			}
214 		}
215 		if (!found) {
216 			throw new QueryErrorException("Query doesn't contain a correct CREATE query:\n" + query);
217 		}
218 		return graphName;
219 	}

222 		final Pattern patternDropGraph = Pattern.compile(
223 				"DROP\\s*(?<silent>SILENT)?\\s*GRAPH\\s*<(?<graph>[^>]*)>",
224 				patternModifier);
225 		Matcher m = patternDropGraph.matcher(query);
226 		boolean found = false;
227 		while (m.find()) {
228 			found = true;
229 			String graphName = m.group("graph");
230 			RevisionManagement.purgeGraph(graphName);
231 		}
232 		if (!found) {
233 			throw new QueryErrorException("Query contain errors:\n" + query);
234 		}

237 	public static void sparqlTagOrBranch(final String sparqlQuery, final String user, final String commitMessage)
238 			throws InternalErrorException, QueryErrorException {
239 		final Pattern patternBranchOrTagQuery = Pattern.compile(
240 				"(?<action>TAG|BRANCH)\\s*GRAPH\\s*<(?<graph>[^>]*)>\\s*REVISION\\s*\"(?<revision>[^\"]*)\"\\s*TO\\s*\"(?<name>[^\"]*)\"",
241 				patternModifier);
242 		Matcher m = patternBranchOrTagQuery.matcher(sparqlQuery);
243 
244 		boolean foundEntry = false;
245 		while (m.find()) {
246 			foundEntry = true;
247 			String action = m.group("action");
248 			String graphName = m.group("graph");
249 			String revisionNumber = m.group("revision").toLowerCase();
250 			String referenceName = m.group("name").toLowerCase();
251 			if (action.equals("TAG")) {
252 				RevisionManagement.createTag(graphName, revisionNumber, referenceName, user, commitMessage);
253 			} else if (action.equals("BRANCH")) {
254 				RevisionManagement.createBranch(graphName, revisionNumber, referenceName, user, commitMessage);
255 			} else {
256 				throw new QueryErrorException("Error in query: " + sparqlQuery);
257 			}	    
258 		}
259 		if (!foundEntry) {
260 			throw new QueryErrorException("Error in query: " + sparqlQuery);
261 		}
262 	}

264 	public static MergeResult sparqlMerge(final String sparqlQuery, final String user, final String commitMessage, final String format) throws InternalErrorException {
265 		final Pattern patternMergeQuery =  Pattern.compile(
266 				"MERGE\\s*(?<action>AUTO|MANUAL)?\\s*GRAPH\\s*<(?<graph>[^>]*?)>\\s*(\\s*(?<sdd>SDD)?\\s*<(?<sddURI>[^>]*?)>)?\\s*BRANCH\\s*\"(?<branchNameA>[^\"]*?)\"\\s*INTO\\s*\"(?<branchNameB>[^\"]*?)\"(\\s*(?<with>WITH)?\\s*\\{(?<triples>.*)\\})?",
267 				patternModifier);
268 		Matcher m = patternMergeQuery.matcher(sparqlQuery);
269 		
270 		if (!m.find())
271 			throw new InternalErrorException("Error in query: " + sparqlQuery);
272 		
273 		String action = m.group("action");
274 		String graphName = m.group("graph");
275 		String sdd = m.group("sdd");
276 		String sddURI = m.group("sddURI");
277 		String branchNameA = m.group("branchNameA").toLowerCase();
278 		String branchNameB = m.group("branchNameB").toLowerCase();
279 		String with = m.group("with");
280 		String triples = m.group("triples");
281 		
282 		String revisionUriA = RevisionManagement.getRevisionUri(graphName, branchNameA);
283 		String revisionUriB = RevisionManagement.getRevisionUri(graphName, branchNameB);
284 		
285 		logger.debug("action: " + action);
286 		logger.debug("graph: " + graphName);
287 		logger.debug("sdd: " + sdd);
288 		logger.debug("sddURI: " + sddURI);
289 		logger.debug("branchNameA: " + branchNameA);
290 		logger.debug("branchNameB: " + branchNameB);
291 		logger.debug("with: " + with);
292 		logger.debug("triples: " + triples);
293 		
294 		MergeResult mresult = new MergeResult(graphName, branchNameA, branchNameB);
295 		
296 		if (!RevisionManagement.checkGraphExistence(graphName)){
297 			logger.error("Graph <"+graphName+"> does not exist.");
298 			throw new InternalErrorException("Graph <"+graphName+"> does not exist.");
299 		}
300 			
301 		
302 		// Check if A and B are different revisions
303 		if (RevisionManagement.getRevisionNumber(graphName, branchNameA).equals(RevisionManagement.getRevisionNumber(graphName, branchNameB))) {
304 			// Branches are equal - throw error
305 			throw new InternalErrorException("Specified branches are equal: " + sparqlQuery);
306 		}
307 		
308 		// Check if both are terminal nodes
309 		if (!(RevisionManagement.isBranch(graphName, branchNameA) && RevisionManagement.isBranch(graphName, branchNameB))) {
310 			throw new InternalErrorException("Non terminal nodes were used: " + sparqlQuery);
311 		}
312 
313 		
314 		
315 		
316 		// Differ between MERGE query with specified SDD and without SDD			
317 		String usedSDDURI = null;
318 		if (sdd != null) {
319 			// Specified SDD
320 			usedSDDURI = sddURI;
321 		} else {
322 			// Default SDD
323 			// Query the referenced SDD
324 			String querySDD = String.format(
325 					  "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
326 					+ "PREFIX rmo: <http://eatld.et.tu-dresden.de/rmo#> %n"
327 					+ "SELECT ?defaultSDD %n"
328 					+ "WHERE { GRAPH <%s> {	%n"
329 					+ "	<%s> a rmo:Graph ;%n"
330 					+ "		sddo:hasDefaultSDD ?defaultSDD . %n"
331 					+ "} }", Config.revision_graph, graphName);
332 			
333 			ResultSet resultSetSDD = TripleStoreInterfaceSingleton.get().executeSelectQuery(querySDD);
334 			if (resultSetSDD.hasNext()) {
335 				QuerySolution qs = resultSetSDD.next();
336 				usedSDDURI = qs.getResource("?defaultSDD").toString();
337 			} else {
338 				throw new InternalErrorException("Error in revision graph! Selected graph <" + graphName + "> has no default SDD referenced.");
339 			}
340 		}
341 
342 		// Get the common revision with shortest path
343 		String commonRevision = MergeManagement.getCommonRevisionWithShortestPath(revisionUriA, revisionUriB);
344 		
345 		// Create the revision progress for A and B
346 		String graphNameA = graphName + "-RM-REVISION-PROGRESS-A";
347 		String graphNameB = graphName + "-RM-REVISION-PROGRESS-B";
348 		String graphNameDiff = graphName + "-RM-DIFFERENCE-MODEL";
349 		mresult.graphDiff = graphNameDiff;
350 		String uriA = "http://eatld.et.tu-dresden.de/branch-A";
351 		String uriB = "http://eatld.et.tu-dresden.de/branch-B";
352 		
353 		MergeManagement.createRevisionProgresses(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriA), graphNameA, uriA, MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriB), graphNameB, uriB);
354 		
355 		// Create difference model
356 		MergeManagement.createDifferenceTripleModel(graphName,  graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI);
357 		
358 		// Differ between the different merge queries
359 		if ((action != null) && (action.equalsIgnoreCase("AUTO")) && (with == null) && (triples == null)) {
360 			logger.info("AUTO MERGE query detected");
361 			// Create the merged revision
362 			mresult.newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.AUTO, "");
363 		} else if ((action != null) && (action.equalsIgnoreCase("MANUAL")) && (with != null) && (triples != null)) {
364 			logger.info("MANUAL MERGE query detected");
365 			// Create the merged revision
366 			mresult.newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.MANUAL, triples);
367 		} else if ((action == null) && (with != null) && (triples != null)) {
368 			logger.info("MERGE WITH query detected");
369 			// Create the merged revision
370 			mresult.newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.WITH, triples);
371 		} else if ((action == null) && (with == null) && (triples == null)) {
372 			logger.info("MERGE query detected");
373 			// Check if difference model contains conflicts
374 			String queryASK = String.format(
375 					  "ASK { %n"
376 					+ "	GRAPH <%s> { %n"
377 					+ " 	?ref <http://eatld.et.tu-dresden.de/sddo#isConflicting> \"true\"^^<http://www.w3.org/2001/XMLSchema#boolean> . %n"
378 					+ "	} %n"
379 					+ "}", graphNameDiff);
380 			if (TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK)) {
381 				// Difference model contains conflicts
382 				// Return the conflict model to the client
383 				mresult.hasConflict = true;
384 				mresult.conflictModel = RevisionManagement.getContentOfGraphByConstruct(graphNameDiff, format);
385 				
386 			} else {
387 				// Difference model contains no conflicts
388 				// Create the merged revision
389 				mresult.newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.COMMON, "");
390 			}
391 		} else {
392 			throw new InternalErrorException("This is not a valid MERGE query: " + sparqlQuery);
393 		}
394 		return mresult;
395 		
396 	}
