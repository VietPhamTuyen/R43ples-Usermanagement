3   import org.apache.commons.configuration.ConfigurationException;
4   import org.apache.commons.configuration.PropertiesConfiguration;
5   import org.apache.log4j.Logger;
6   
7   /**
8   * General configuration handling.
9   * Stores all important configuration from R43ples.
10  * Can read from configuration file.
11  *
12  * @author Markus Graube
13  * @author Stephan Hensel
14  *
15  */
16  public class Config {
17  
18  	// Database settings
19  	// Jena
20  	public static String jena_tdb_directory;
21  	
22  	// Virtuoso
23  	public static String virtuoso_url;
24  	public static String virtuoso_user;
25  	public static String virtuoso_password;
26  	
27  	// HTTP
28  	public static String http_url;
29  	public static String http_user;
30  	public static String http_password;
31  	
32  	// Service settings
33  	/** The service host. **/
34  	public static String service_host;
35  	/** The service port. **/
36  	public static int service_port;
37  	/** The service path. **/
38  	public static String service_path;
39  
40  	// SSL settings
41  	/** The SSL keystore path. **/
42  	public static String ssl_keystore;
43  	/** The SSL keystore password. **/
44  	public static String ssl_password;
45  	
46  	// Internal r43ples settings
47  	/** The r43ples revision graph **/
48  	public static String revision_graph;
49  	/** The SDD graph URI. **/
50  	public static String sdd_graph;
51  	/** The path to the SDD graph default content. **/
52  	public static String sdd_graph_defaultContent;
53  	
54  	// Visualisation settings
55  	/** The visualization file path. **/
56  	public static String visualisation_path;
57  	/** The logger. **/
58  	private static Logger logger = Logger.getLogger(Config.class);	
59  	
60  	
61  	/**
62  	* Read the configuration information from local file.
63  	*
64  	* @param configFilePath path to configuration file
65  	* @throws ConfigurationException
66  	*/
67  	public static void readConfig(final String configFilePath) throws ConfigurationException{
68  		PropertiesConfiguration config;
69  		try {
70  			config = new PropertiesConfiguration(configFilePath);
71  			
72  			jena_tdb_directory = config.getString("jena.tdb.directory");
73  			
74  			virtuoso_url = config.getString("virtuoso.url");
75  			virtuoso_user = config.getString("virtuoso.user");
76  			virtuoso_password = config.getString("virtuoso.password");
77  			
78  			http_url = config.getString("http.url");
79  			http_user = config.getString("http.user");
80  			http_password = config.getString("http.password");
81  			
82  			service_host = config.getString("service.host");
83  			service_port = config.getInt("service.port");
84  			service_path = config.getString("service.path");
85  			
86  			ssl_keystore = config.getString("ssl.keystore");
87  			ssl_password = config.getString("ssl.password");
88  			
89  			revision_graph = config.getString("revision.graph");
90  			
91  			sdd_graph = config.getString("sdd.graph");
92  			sdd_graph_defaultContent = config.getString("sdd.graph.defaultContent");
93  
94  			visualisation_path = config.getString("visualisation.path");
95  		} catch (ConfigurationException e) {
96  			logger.warn("Could not read configuration file '" + configFilePath + "'. Switch to 'r43ples.dist.conf'.");
97  			readConfig("r43ples.dist.conf");
98  		}
99  	}
100 
101 }
