46  	 * Get the common revision of the specified revisions which has the shortest path to the two.
47  	 * To ensure wise results the revisions should be terminal branch nodes.
48  	 * 
49  	 * @param revision1 the first revision should be a terminal branch node
50  	 * @param revision2 the second revision should be a terminal branch node
51  	 * @return the nearest common revision
52  	 */
53  	public static String getCommonRevisionWithShortestPath(String revision1, String revision2) {
54  		
55  		logger.info("Get the common revision of <" + revision1 + "> and <" + revision2 + "> which has the shortest path.");
56  		
57  		String query = String.format(
58  			  "PREFIX prov: <http://www.w3.org/ns/prov#> "
59  			  + "SELECT DISTINCT ?revision "
60  			  + "WHERE { "
61  			  + "    GRAPH <%s> {"
62  			  + "        ?branch1 prov:wasDerivedFrom+ ?revision ."
63  			  + "        ?branch2 prov:wasDerivedFrom+ ?revision ."
64  			  + "        FILTER NOT EXISTS {"
65  			  + "            ?next prov:wasDerivedFrom ?revision."
66  			  + "            ?branch1 prov:wasDerivedFrom+ ?next ."
67  			  + "            ?branch2 prov:wasDerivedFrom+ ?next ."
68  			  + "        }"
69  			  + "        FILTER ( ?branch1 = <%s> )"
70  			  + "        FILTER ( ?branch2 = <%s> )"
71  			  + "    }"
72  			  + "}"
73  			  + "LIMIT 1",
74  			  Config.revision_graph, revision1, revision2);
75  		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
76  		
77  		if (results.hasNext()) {
78  			QuerySolution qs = results.next();
79  			logger.info("Common revision found.");
80  			return qs.getResource("?revision").toString();
81  		}
82  		
83  		logger.info("No common revision could be found.");
84  		return null;		
85  	}

88  	/**
89  	 * Calculate the path from start revision to target revision.
90  	 * 
91  	 * @param startRevision the start revision
92  	 * @param targetRevision the target revision
93  	 * @return linked list with all revisions from start revision to target revision
94  	 */
95  	public static LinkedList<String> getPathBetweenStartAndTargetRevision(String startRevision, String targetRevision) {
96  		
97  		logger.info("Calculate the shortest path from revision <" + startRevision + "> to <" + targetRevision + "> .");
98  		String query = String.format(
99  			  "PREFIX prov: <http://www.w3.org/ns/prov#> %n"
100 			+ "SELECT DISTINCT ?revision ?previousRevision %n"
101 			+ "WHERE { %n"
102 			+ "	GRAPH <%s> { %n"
103 			+ "		<%s> prov:wasDerivedFrom* ?revision."
104 			+ "		?revision prov:wasDerivedFrom* <%s>."
105 			+ "		OPTIONAL{?revision prov:wasDerivedFrom ?previousRevision}"
106 			+ " }"
107 			+ "}", Config.revision_graph, targetRevision, startRevision);
108 		
109 		HashMap<String, ArrayList<String>> resultMap = new HashMap<String, ArrayList<String>>();
110 		LinkedList<String> list = new LinkedList<String>();
111 		
112 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
113 
114 		// Path element counter
115 		int counterLength = 0;
116 		
117 		while (resultSet.hasNext()) {
118 			QuerySolution qs = resultSet.next();
119 			String resource = qs.getResource("?revision").toString();
120 			String previousResource = null;
121 			if (qs.getResource("?previousRevision") != null) {
122 				previousResource = qs.getResource("?previousRevision").toString();
123 			}
124 			if (resultMap.containsKey(resource)) {
125 				resultMap.get(resource).add(previousResource);
126 			} else {
127 				ArrayList<String> arrayList = new ArrayList<String>();
128 				counterLength++;
129 				arrayList.add(previousResource);
130 				resultMap.put(resource, arrayList);
131 			}
132 		}
133 		
134 		// Sort the result map -> sorted list of path elements
135 		// A merged revision can have two predecessors -> it is important to choose the right predecessor revision according to the selected path
136 		String currentPathElement = targetRevision;
137 		for (int i = 0; i < counterLength; i++) {
138 			list.addFirst(currentPathElement);
139 			
140 			// Check if start revision was already reached
141 			if (currentPathElement.equals(startRevision)) {
142 				return list;
143 			}
144 			
145 			if (resultMap.get(currentPathElement).size() > 1) {
146 				if (resultMap.containsKey(resultMap.get(currentPathElement).get(0))) {
147 					currentPathElement = resultMap.get(currentPathElement).get(0);
148 				} else {
149 					currentPathElement = resultMap.get(currentPathElement).get(1);
150 				}
151 			} else {
152 				currentPathElement = resultMap.get(currentPathElement).get(0);
153 			}
154 		}
155 
156 		return list;
157 	}

172 		logger.info("Create the revision progress of branch A and B.");
173 		
174 		// Get the common revision
175 		String commonRevision = null;
176 		if ((listA.size() > 0) && (listB.size() > 0)) {
177 			commonRevision = listA.getFirst();
178 		} else {
179 			throw new InternalErrorException("Revision path contains no revisions.");
180 		}
181 
182 		// Get the revision number of first revision
183 		logger.info("Get the revision number of first revision.");
184 		String firstRevisionNumber = "";
185 		String graphName = "";
186 
187 		String query = String.format(
188 			  "SELECT ?number ?graph %n"
189 			+ "WHERE { %n"
190 			+ "	GRAPH <%s> { %n"
191 			+ "		<%s> <http://eatld.et.tu-dresden.de/rmo#revisionNumber> ?number . %n"
192 			+ " 	<%s> <http://eatld.et.tu-dresden.de/rmo#revisionOf> ?graph . %n"
193 			+ "} }", Config.revision_graph, commonRevision, commonRevision);
194 		
195 		ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
196 		
197 		if (results.hasNext()) {
198 			QuerySolution qs = results.next();
199 			firstRevisionNumber = qs.getLiteral("?number").toString();
200 			graphName = qs.getResource("?graph").toString();
201 		}
202 		
203 		// Get the full graph name of first revision or create full revision graph of first revision
204 		String fullGraphNameCommonRevision = "";
205 		Boolean tempGraphWasCreated = false;
206 		try {
207 			fullGraphNameCommonRevision = RevisionManagement.getReferenceGraph(graphName, firstRevisionNumber);
208 		} catch (InternalErrorException e) {
209 			// Create a temporary full graph
210 			fullGraphNameCommonRevision = graphName + "RM-TEMP-REVISION-PROGRESS-FULLGRAPH";
211 			RevisionManagement.generateFullGraphOfRevision(graphName, firstRevisionNumber, fullGraphNameCommonRevision);
212 			tempGraphWasCreated = true;
213 		}
214 		
215 		// Create revision progress of branch A
216 		createRevisionProgress(listA, fullGraphNameCommonRevision, graphNameRevisionProgressA, uriA);
217 		
218 		// Create revision progress of branch A
219 		createRevisionProgress(listB, fullGraphNameCommonRevision, graphNameRevisionProgressB, uriB);
220 		
221 		// Drop the temporary full graph
222 		if (tempGraphWasCreated) {
223 			logger.info("Drop the temporary full graph.");
224 			TripleStoreInterfaceSingleton.get().executeUpdateQuery("DROP SILENT GRAPH <" + fullGraphNameCommonRevision + ">");
225 		}
226

497 		String sparqlTemplateRevisionB = String.format(
498 				  "	GRAPH <%s> { %n"
499 				+ "		<%s> <%s> ?blankB . %n"
500 				+ "			?blankB rdf:subject ?s . %n"
501 				+ "			?blankB rdf:predicate ?p . %n"
502 				+ "			?blankB rdf:object ?o . %n"
503 				+ "			?blankB rmo:references ?revisionB . %n"
504 				+ "	} %n", graphNameRevisionProgressB, uriB, "%s");
505 
506 		String sparqlTemplateNotExistsRevisionA = String.format(
507 				  "FILTER NOT EXISTS { %n"
508 				+ "	GRAPH <%s> { %n"
509 				+ "		<%s> ?everything ?blankA . %n"
510 				+ "			?blankA rdf:subject ?s . %n"
511 				+ "			?blankA rdf:predicate ?p . %n"
512 				+ "			?blankA rdf:object ?o . %n"
513 				+ "			?blankA rmo:references ?revisionA . %n"
514 				+ "	} %n"
515 				+ "}", graphNameRevisionProgressA, uriA);
516 		
517 		String sparqlTemplateNotExistsRevisionB = String.format(
518 				  "FILTER NOT EXISTS { %n"
519 				+ "	GRAPH <%s> { %n"
520 				+ "		<%s> ?everything ?blankB . %n"
521 				+ "			?blankB rdf:subject ?s . %n"
522 				+ "			?blankB rdf:predicate ?p . %n"
523 				+ "			?blankB rdf:object ?o . %n"
524 				+ "			?blankB rmo:references ?revisionB . %n"
525 				+ "	} %n"
526 				+ "}", graphNameRevisionProgressB, uriB);
527 		
528 		// Get all structural definitions which are generating differences
529 		String queryDifferingSD = String.format(
530 				  "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
531 				+ "PREFIX sdd:  <http://eatld.et.tu-dresden.de/sdd#> %n"
532 				+ "PREFIX xsd:  <http://www.w3.org/2001/XMLSchema#> %n"
533 				+ "SELECT ?combinationURI ?tripleStateA ?tripleStateB ?conflict ?automaticResolutionState %n"
534 				+ "WHERE { GRAPH <%s> { %n"
535 				+ "	<%s> a sddo:StructuralDefinitionGroup ;"
536 				+ "		sddo:hasStructuralDefinition ?combinationURI ."
537 				+ "	?combinationURI a sddo:StructuralDefinition ; %n"
538 				+ "		sddo:hasTripleStateA ?tripleStateA ; %n"
539 				+ "		sddo:hasTripleStateB ?tripleStateB ; %n"
540 				+ "		sddo:isConflicting ?conflict ; %n"
541 				+ "		sddo:automaticResolutionState ?automaticResolutionState . %n"
542 				+ "} } %n", Config.sdd_graph, uriSDD);
543 				
544 		// Iterate over all differing combination URIs
545 		ResultSet resultSetDifferences = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryDifferingSD);
546 		while (resultSetDifferences.hasNext()) {
547 			QuerySolution qs = resultSetDifferences.next();
548 
549 			String currentDifferenceCombinationURI = qs.getResource("?combinationURI").toString();
550 			String currentTripleStateA = qs.getResource("?tripleStateA").toString();
551 			String currentTripleStateB = qs.getResource("?tripleStateB").toString();
552 			// Will return an integer value because virtuoso stores boolean internal as integer
553 			String currentConflictState = qs.getLiteral("?conflict").toString();
554 			// TDB returns boolean value without "" -> add it to use it in the next query correctly
555 			if (currentConflictState.equals("true^^http://www.w3.org/2001/XMLSchema#boolean")) {
556 				currentConflictState = "\"true\"^^<http://www.w3.org/2001/XMLSchema#boolean>";
557 			} else {
558 				currentConflictState = "\"false\"^^<http://www.w3.org/2001/XMLSchema#boolean>";
559 			}
560 			String currentAutomaticResolutionState = qs.getResource("?automaticResolutionState").toString();
561 			
562 			String querySelectPart = "SELECT ?s ?p ?o %s %s %n";
563 			String sparqlQueryRevisionA = null;
564 			String sparqlQueryRevisionB = null;			
565 			
566 			// A
567 			if (currentTripleStateA.equals(SDDTripleState.ADDED.getSddRepresentation())) {
568 				// In revision A the triple was added
569 				querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
570 				sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleState.ADDED.getRpoRepresentation());
571 			} else if (currentTripleStateA.equals(SDDTripleState.DELETED.getSddRepresentation())) {
572 				// In revision A the triple was deleted
573 				querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
574 				sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleState.DELETED.getRpoRepresentation());
575 			} else if (currentTripleStateA.equals(SDDTripleState.ORIGINAL.getSddRepresentation())) {
576 				// In revision A the triple is original
577 				querySelectPart = String.format(querySelectPart, "?revisionA", "%s");
578 				sparqlQueryRevisionA = String.format(sparqlTemplateRevisionA, SDDTripleState.ORIGINAL.getRpoRepresentation());
579 			} else if (currentTripleStateA.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation())) {
580 				// In revision A the triple is not included
581 				querySelectPart = String.format(querySelectPart, "", "%s");
582 				sparqlQueryRevisionA = sparqlTemplateNotExistsRevisionA;
583 			}
584 			
585 			// B
586 			if (currentTripleStateB.equals(SDDTripleState.ADDED.getSddRepresentation())) {
587 				// In revision B the triple was added
588 				querySelectPart = String.format(querySelectPart, "?revisionB");
589 				sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleState.ADDED.getRpoRepresentation());
590 			} else if (currentTripleStateB.equals(SDDTripleState.DELETED.getSddRepresentation())) {
591 				// In revision B the triple was deleted
592 				querySelectPart = String.format(querySelectPart, "?revisionB");
593 				sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleState.DELETED.getRpoRepresentation());
594 			} else if (currentTripleStateB.equals(SDDTripleState.ORIGINAL.getSddRepresentation())) {
595 				// In revision B the triple is original
596 				querySelectPart = String.format(querySelectPart, "?revisionB");
597 				sparqlQueryRevisionB = String.format(sparqlTemplateRevisionB, SDDTripleState.ORIGINAL.getRpoRepresentation());
598 			} else if (currentTripleStateB.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation())) {
599 				// In revision B the triple is not included
600 				querySelectPart = String.format(querySelectPart, "");
601 				sparqlQueryRevisionB = sparqlTemplateNotExistsRevisionB;
602 			}
603 		
604 			// Concatenated SPARQL query
605 			String query = String.format(
606 					prefixes
607 					+ "%s"
608 					+ "WHERE { %n"
609 					+ "%s"
610 					+ "%s"
611 					+ "} %n", querySelectPart, sparqlQueryRevisionA, sparqlQueryRevisionB);
612 					
613 			// Iterate over all triples
614 			ResultSet resultSetTriples = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
615 			while (resultSetTriples.hasNext()) {
616 				QuerySolution qsQuery = resultSetTriples.next();
617 				
618 				String subject = qsQuery.getResource("?s").toString();
619 				String predicate = qsQuery.getResource("?p").toString();
620 
621 				// Differ between literal and resource
622 				String object = "";
623 				if (qsQuery.get("?o").isLiteral()) {
624 					object = "\"" + qsQuery.getLiteral("?o").toString() + "\"";
625 				} else {
626 					object = "<" + qsQuery.getResource("?o").toString() + ">";
627 				}
628 				
629 				// Create the references A and B part of the query
630 				String referencesAB = ". %n";
631 				if (!currentTripleStateA.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation()) && !currentTripleStateB.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation())) {
632 					referencesAB = String.format(
633 							  "			rpo:referencesA <%s> ; %n"
634 							+ "			rpo:referencesB <%s> %n", qsQuery.getResource("?revisionA").toString(), 
635 																qsQuery.getResource("?revisionB").toString());
636 				} else if (currentTripleStateA.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation()) && !currentTripleStateB.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation())) {
637 					referencesAB = String.format(
638 							  "			rpo:referencesB <%s> %n", qsQuery.getResource("?revisionB").toString());
639 				} else if (!currentTripleStateA.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation()) && currentTripleStateB.equals(SDDTripleState.NOTINCLUDED.getSddRepresentation())) {
640 					referencesAB = String.format(
641 							  "			rpo:referencesA <%s> %n", qsQuery.getResource("?revisionA").toString());
642 				}
643 				
644 				String queryTriple = prefixes + String.format(
645 						  "INSERT DATA { GRAPH <%s> {%n"
646 						+ "	<%s> a rpo:DifferenceGroup ; %n"
647 						+ "	sddo:hasTripleStateA <%s> ; %n"
648 						+ "	sddo:hasTripleStateB <%s> ; %n"
649 						+ "	sddo:isConflicting %s ; %n"
650 						+ "	sddo:automaticResolutionState <%s> ; %n"
651 						+ "	rpo:hasDifference [ %n"
652 						+ "		a rpo:Difference ; %n"
653 						+ "			rpo:hasTriple [ %n"
654 						+ "				rdf:subject <%s> ; %n"
655 						+ "				rdf:predicate <%s> ; %n"
656 						+ "				rdf:object %s %n"
657 						+ "			] ; %n"
658 						+ "%s"
659 						+ "	] . %n"
660 						+ "} }", graphNameDifferenceTripleModel, 
661 									currentDifferenceCombinationURI, 
662 									currentTripleStateA, 
663 									currentTripleStateB,
664 									currentConflictState,
665 									currentAutomaticResolutionState,
666 									subject, 
667 									predicate,
668 									object,
669 									referencesAB);
670 				
671 				TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryTriple);
672 			}
673 		}

677 	/**
678 	 * Create a merged revision.
679 	 * 
680 	 * @param graphName the graph name
681 	 * @param branchNameA the name of branch A
682 	 * @param branchNameB the name of branch B
683 	 * @param user the user
684 	 * @param commitMessage the commit message
685 	 * @param graphNameDifferenceTripleModel the graph name of the difference triple model
686 	 * @param graphNameRevisionProgressA the graph name of the revisions progress A
687 	 * @param uriA the URI A
688 	 * @param graphNameRevisionProgressB the graph name of the revisions progress B
689 	 * @param uriB the URI B
690 	 * @param uriSDD the URI of the SDD
691 	 * @param type the merge query type
692 	 * @param triples the triples which are belonging to the current merge query in N-Triple serialization
693 	 * @return new revision number
694 	 * @throws InternalErrorException 
695 	 */
696 	public static String createMergedRevision(String graphName, String branchNameA, String branchNameB, String user, String commitMessage, String graphNameDifferenceTripleModel, String graphNameRevisionProgressA, String uriA, String graphNameRevisionProgressB, String uriB, String uriSDD, MergeQueryTypeEnum type, String triples) throws InternalErrorException {
697 		 
698 		// Create an empty temporary graph which will contain the merged full content
699 		String graphNameOfMerged = graphName + "-RM-MERGED-TEMP";
700 		createNewGraph(graphNameOfMerged);
701 		
702 		// Get the full graph name of branch A
703 		String graphNameOfBranchA = RevisionManagement.getReferenceGraph(graphName, branchNameA);
704 		// Get the full graph name of branch B
705 		String graphNameOfBranchB = RevisionManagement.getReferenceGraph(graphName, branchNameB);
706 		
707 		if (type.equals(MergeQueryTypeEnum.MANUAL)) {
708 			// Manual merge query
709 			RevisionManagement.executeINSERT(graphNameOfMerged, triples);
710 		} else {	
711 			// Copy graph B to temporary merged graph
712 			String queryCopy = String.format("COPY <%s> TO <%s>", graphNameOfBranchB, graphNameOfMerged);
713 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryCopy);
714 			
715 			// Get the triples from branch A which should be added to/removed from the merged revision
716 			String triplesToAdd = "";
717 			String triplesToDelete = "";
718 			
719 			// Get all difference groups
720 			String queryDifferenceGroup = prefixes + String.format(
721 					  "SELECT ?differenceCombinationURI ?automaticResolutionState ?tripleStateA ?tripleStateB ?conflict %n"
722 					+ "WHERE { GRAPH <%s> { %n"
723 					+ "	?differenceCombinationURI a rpo:DifferenceGroup ; %n"
724 					+ "		sddo:automaticResolutionState ?automaticResolutionState ; %n"
725 					+ "		sddo:hasTripleStateA ?tripleStateA ; %n"
726 					+ "		sddo:hasTripleStateB ?tripleStateB ; %n"
727 					+ "		sddo:isConflicting ?conflict . %n"
728 					+ "} }", graphNameDifferenceTripleModel);
729 	
730 			// Iterate over all difference groups
731 			ResultSet resultSetDifferenceGroups = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryDifferenceGroup);
732 			while (resultSetDifferenceGroups.hasNext()) {
733 				QuerySolution qsCurrentDifferenceGroup = resultSetDifferenceGroups.next();
734 	
735 				String currentDifferencGroupURI = qsCurrentDifferenceGroup.getResource("?differenceCombinationURI").toString();
736 				String currentDifferencGroupAutomaticResolutionState = qsCurrentDifferenceGroup.getResource("?automaticResolutionState").toString();
737 //				Currently not needed
738 //				String currentDifferencGroupTripleStateA = qsCurrentDifferenceGroup.getResource("?tripleStateA").toString();
739 //				String currentDifferencGroupTripleStateB = qsCurrentDifferenceGroup.getResource("?tripleStateB").toString();
740 				boolean currentDifferencGroupConflict = qsCurrentDifferenceGroup.getLiteral("?conflict").getBoolean();
741 				
742 				// Get all differences (triples) of current difference group
743 				String queryDifference = prefixes + String.format(
744 						  "SELECT ?s ?p ?o %n"
745 						+ "WHERE { GRAPH <%s> { %n"
746 						+ "	<%s> a rpo:DifferenceGroup ; %n"
747 						+ "		rpo:hasDifference ?blankDifference . %n"
748 						+ "	?blankDifference a rpo:Difference ; %n"
749 						+ "		rpo:hasTriple ?triple . %n"
750 						+ "	?triple rdf:subject ?s . %n"
751 						+ "	?triple rdf:predicate ?p . %n"
752 						+ "	?triple rdf:object ?o . %n"
753 						+ "} }", graphNameDifferenceTripleModel, currentDifferencGroupURI);
754 				
755 				// Iterate over all differences (triples)
756 				ResultSet resultSetDifferences = TripleStoreInterfaceSingleton.get().executeSelectQuery(queryDifference);
757 				while (resultSetDifferences.hasNext()) {
758 					QuerySolution qsCurrentDifference = resultSetDifferences.next();
759 					
760 					String subject = "<" + qsCurrentDifference.getResource("?s").toString() + ">";
761 					String predicate = "<" + qsCurrentDifference.getResource("?p").toString() + ">";
762 	
763 					// Differ between literal and resource
764 					String object = "";
765 					if (qsCurrentDifference.get("?o").isLiteral()) {
766 						object = "\"" + qsCurrentDifference.getLiteral("?o").toString() + "\"";
767 					} else {
768 						object = "<" + qsCurrentDifference.getResource("?o").toString() + ">";
769 					}
770 					
771 					if (	type.equals(MergeQueryTypeEnum.AUTO) || 
772 							type.equals(MergeQueryTypeEnum.COMMON) || 
773 							(type.equals(MergeQueryTypeEnum.WITH) && !currentDifferencGroupConflict) ) {
774 						// MERGE AUTO or common MERGE query
775 						if (currentDifferencGroupAutomaticResolutionState.equals(SDDTripleState.ADDED.getSddRepresentation())) {
776 							// Triple should be added
777 							triplesToAdd += subject + " " + predicate + " " + object + " . \n";
778 						} else {
779 							// Triple should be deleted
780 							triplesToDelete += subject + " " + predicate + " " + object + " . \n";
781 						}
782 					} else {
783 						// MERGE WITH query - conflicting triple
784 						Model model = JenaModelManagement.readNTripleStringToJenaModel(triples);
785 						// Create ASK query which will check if the model contains the specified triple
786 						String queryAsk = String.format(
787 								  "ASK { %n"
788 								+ " %s %s %s %n"
789 								+ "}", subject, predicate, object);
790 						Query query = QueryFactory.create(queryAsk);
791 						QueryExecution qe = QueryExecutionFactory.create(query, model);
792 						boolean resultAsk = qe.execAsk();
793 						qe.close();
794 						model.close();
795 						if (resultAsk) {
796 							// Model contains the specified triple
797 							// Triple should be added
798 							triplesToAdd += subject + " " + predicate + " " + object + " . \n";
799 						} else {
800 							// Triple should be deleted
801 							triplesToDelete += subject + " " + predicate + " " + object + " . \n";
802 						}
803 					}
804 				}
805 				// Update the merged graph
806 				// Insert triplesToAdd
807 				RevisionManagement.executeINSERT(graphNameOfMerged, triplesToAdd);
808 				// Delete triplesToDelete
809 				RevisionManagement.executeDELETE(graphNameOfMerged, triplesToDelete);
810 			}
811 		}
812 		
813 		// Calculate the add and delete sets
814 		
815 		// Get all added triples (concatenate all triples which are in MERGED but not in A and all triples which are in MERGED but not in B)
816 		String queryAddedTriples = String.format(
817 				  "CONSTRUCT {?s ?p ?o} %n"
818 				+ "WHERE { %n"
819 				+ "	GRAPH <%s> { ?s ?p ?o } %n"
820 				+ "	FILTER NOT EXISTS { %n"
821 				+ "		GRAPH <%s> { ?s ?p ?o } %n"
822 				+ "	} %n"
823 				+ "}", graphNameOfMerged, graphNameOfBranchA);
824 		
825 		String addedTriples = TripleStoreInterfaceSingleton.get().executeConstructQuery(queryAddedTriples, FileUtils.langNTriple);
826 		
827 		queryAddedTriples = String.format(
828 				  "CONSTRUCT {?s ?p ?o} %n"
829 				+ "WHERE { %n"
830 				+ "	GRAPH <%s> { ?s ?p ?o } %n"
831 				+ "	FILTER NOT EXISTS { %n"
832 				+ "		GRAPH <%s> { ?s ?p ?o } %n"
833 				+ "	} %n"
834 				+ "}", graphNameOfMerged, graphNameOfBranchB);
835 
836 		addedTriples += TripleStoreInterfaceSingleton.get().executeConstructQuery(queryAddedTriples, FileUtils.langNTriple);
837 		
838 		// Get all removed triples (concatenate all triples which are in A but not in MERGED and all triples which are in B but not in MERGED)
839 		String queryRemovedTriples = String.format(
840 				  "CONSTRUCT {?s ?p ?o} %n"
841 				+ "WHERE { %n"
842 				+ "	GRAPH <%s> { ?s ?p ?o } %n"
843 				+ "	FILTER NOT EXISTS { %n"
844 				+ "		GRAPH <%s> { ?s ?p ?o } %n"
845 				+ "	} %n"
846 				+ "}", graphNameOfBranchA, graphNameOfMerged);
847 		
848 		String removedTriples = TripleStoreInterfaceSingleton.get().executeConstructQuery(queryRemovedTriples, FileUtils.langNTriple);
849 		
850 		queryRemovedTriples = String.format(
851 				  "CONSTRUCT {?s ?p ?o} %n"
852 				+ "WHERE { %n"
853 				+ "	GRAPH <%s> { ?s ?p ?o } %n"
854 				+ "	FILTER NOT EXISTS { %n"
855 				+ "		GRAPH <%s> { ?s ?p ?o } %n"
856 				+ "	} %n"
857 				+ "}", graphNameOfBranchB, graphNameOfMerged);
858 		
859 		removedTriples += TripleStoreInterfaceSingleton.get().executeConstructQuery(queryRemovedTriples, FileUtils.langNTriple);
860 
861 		// Create list with the 2 predecessors - the order is important - fist item will specify the branch were the new merged revision will be created
862 		ArrayList<String> usedRevisionNumbers = new ArrayList<String>();
863 		usedRevisionNumbers.add(branchNameB);
864 		usedRevisionNumbers.add(branchNameA);
865 		return RevisionManagement.createNewRevision(graphName, addedTriples, removedTriples, user, commitMessage, usedRevisionNumbers);
866 	}

872 	 * 
873 	 * @param graphname the graph name
874 	 */
875 	private static void createNewGraph(String graphName) {
876 		logger.info("Create new graph with the name: " + graphName + ".");

878 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format("CREATE GRAPH  <%s>", graphName));
879 	}
