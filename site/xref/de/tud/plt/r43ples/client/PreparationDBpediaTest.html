13  public class PreparationDBpediaTest {
14  
15  	private static Logger logger = Logger
16  			.getLogger(PreparationDBpediaTest.class);
17  
18  	public static void main(String[] args) throws ConfigurationException, InternalErrorException {
19  		logger.info("Create DBpedia dataset");
20  		
21  		String path = args[0];
22  		
23  		String graph_name = "http://dbpedia.org";
24  		
25  		Config.readConfig("r43ples.conf");
26  		TripleStoreInterfaceFactory.createInterface();
27  
28  		// load instance data to triplestore (dbpedia_2013_07_18.nt)
29  		String file_name = "dbpedia_2013_07_18.nt";
30  		String command = String.format("ld_dir('%s', '%s', '%s');%n", path, file_name, graph_name);
31  
32  		// create revision information for instance data
33  		RevisionManagement.putGraphUnderVersionControl(graph_name);
34  		
35  		String user = "dbpedia-test";
36  		
37  		// insert changesets into R43ples
38  		final int MAX_REV = 277;
39  		for (int i=0; i <= MAX_REV; i++){
40  			logger.info("Add revision to dbpedia: "+i);
41  			String newRevisionNumber = String.format("%d",i +1);
42  			String addSetGraphUri = graph_name + "-delta-added-" + newRevisionNumber;
43  			String removeSetGraphUri = graph_name + "-delta-removed-" + newRevisionNumber;
44  			ArrayList<String> list = new ArrayList<String>();
45  			list.add(Integer.toString(i));
46  			command += String.format("ld_dir('%s', '%06d.added.nt', '%s');%n", path, i, addSetGraphUri );
47  			command += String.format("ld_dir('%s', '%06d.removed.nt', '%s');%n", path, i, removeSetGraphUri );
48  			
49  			RevisionManagement.addMetaInformationForNewRevision(graph_name, user, "create revision " + newRevisionNumber, list,
50  					newRevisionNumber, addSetGraphUri, removeSetGraphUri);	
51  		}
52  		
53  		command += "select * from DB.DBA.load_list;\n"
54  				+ "rdf_loader_run();";
55  		logger.info("Command to be executed on ISQL interface in Virtuoso:\n" + command);
56  	}
57  
58  }
