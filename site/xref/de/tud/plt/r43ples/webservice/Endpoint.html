408 	/**
409 	 * @param format
410 	 * 			requested mime type 
411 	 * @param sparqlQuery
412 	 * 			string containing the SPARQL query
413 	 * @return HTTP response of evaluating the sparql query 
414 	 * @throws InternalErrorException
415 	 */
416 	private Response getSparqlResponse(final String format, String sparqlQuery, final String join_option) throws InternalErrorException {
417 		logger.info(String.format("SPARQL request (format=%s, join_option=%s) -> %n %s", format, join_option, sparqlQuery));
418 		String user = null;
419 		Matcher userMatcher = patternUser.matcher(sparqlQuery);
420 		if (userMatcher.find()) {
421 			user = userMatcher.group("user");
422 			sparqlQuery = userMatcher.replaceAll("");
423 		}
424 		String message = null;
425 		Matcher messageMatcher = patternCommitMessage.matcher(sparqlQuery);
426 		if (messageMatcher.find()) {
427 			message = messageMatcher.group("message");
428 			sparqlQuery = messageMatcher.replaceAll("");
429 		}
430 
431 		if (patternSelectAskConstructQuery.matcher(sparqlQuery).find()) {
432 			return getSelectConstructAskResponse(sparqlQuery, format, join_option);
433 		}
434 		if (patternUpdateQuery.matcher(sparqlQuery).find()) {
435 			return getUpdateResponse(sparqlQuery, user, message, format);
436 		}
437 		if (patternCreateGraph.matcher(sparqlQuery).find()) {
438 			return getCreateGraphResponse(sparqlQuery, format);
439 		}
440 		if (patternMergeQuery.matcher(sparqlQuery).find()) {
441 			return getMergeResponse(sparqlQuery, user, message, format);
442 		}
443 		if (patternDropGraph.matcher(sparqlQuery).find()) {
444 			return getDropGraphResponse(sparqlQuery, format);
445 		}
446 		if (patternBranchOrTagQuery.matcher(sparqlQuery).find()) {
447 			return getBranchOrTagResponse(sparqlQuery, user, message, format);
448 		}
449 		throw new QueryErrorException("No R43ples query detected");
450 	}

477 	 * Provides the SPARQL Endpoint description of the original sparql endpoint
478 	 * with the additional R43ples feature (sd:feature) and replaced URIs.
479 	 * 
480 	 * @param format
481 	 *            serialisation format of the service description
482 	 * @return Extended Service Description
483 	 * @throws ClientProtocolException
484 	 * @throws IOException
485 	 */
486 	private Response getServiceDescriptionResponse(final String format) {
487 		logger.info("Service Description requested");
488 		String triples =String.format("@prefix rdf:	<http://www.w3.org/1999/02/22-rdf-syntax-ns#> . %n"
489 				+ "@prefix ns3:	<http://www.w3.org/ns/formats/> .%n"
490 				+ "@prefix sd:	<http://www.w3.org/ns/sparql-service-description#> .%n"
491 				+ "<%1$s>	rdf:type	sd:Service ;%n"
492 				+ "	sd:endpoint	<%1$s> ;%n"
493 				+ "	sd:feature	sd:r43ples ;"
494 				+ "	sd:resultFormat	ns3:SPARQL_Results_JSON ,%n"
495 				+ "		ns3:SPARQL_Results_XML ,%n"
496 				+ "		ns3:Turtle ,%n"
497 				+ "		ns3:N-Triples ,%n"
498 				+ "		ns3:N3 ,%n"
499 				+ "		ns3:RDF_XML ,%n"
500 				+ "		ns3:SPARQL_Results_CSV ,%n"
501 				+ "		ns3:RDFa ;%n"
502 				+ "	sd:supportedLanguage	sd:SPARQL10Query, sd:SPARQL11Query, sd:SPARQL11Query, sd:SPARQL11Update, sd:R43plesQuery  ;%n"
503 				+ "	sd:url	<%1$s> .%n", uriInfo.getAbsolutePath()) ;
504 		Model model = JenaModelManagement.readStringToJenaModel(triples, "TURTLE");
505 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
506 		
507 		if (format.toLowerCase().contains("xml") )
508 			model.write(baos, "RDF/XML");
509 		else if (format.toLowerCase().contains("turtle") )
510 			model.write(baos, "Turtle");
511 		else if (format.toLowerCase().contains("json") )
512 			model.write(baos, "RDF/JSON");
513 		else {
514 			try {
515 				model.write(baos, format);
516 			}
517 			catch (NoWriterForLangException e) {
518 				model.write(baos, "Turtle");
519 			}
520 		}
521 		return Response.ok().entity(baos.toString()).build();
522 	}

524 	/**
525 	 * Produce the response for a SELECT, ASK or CONSTRUCT SPARQL query. can handle multiple
526 	 * graphs
527 	 * 
528 	 * @param query
529 	 *            the SPARQL query
530 	 * @param format
531 	 *            the result format
532 	 * @param join_option
533 	 * 			   use inner JIOn option for query
534 	 * @return the response with HTTP header for every graph (revision number
535 	 *         and MASTER revision number)
536 	 * @throws InternalErrorException 
537 	 */
538 	private Response getSelectConstructAskResponse(final String query, final String format, final String join_option) throws InternalErrorException {
539 		ResponseBuilder responseBuilder = Response.ok();
540 		String result;
541 		
542 		if (join_option.equals("old")) {
543 			String query_rewritten = SparqlRewriter_old.rewriteQuery(query);
544 			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
545 		} 
546 		else if (join_option.equals("new")) {
547 			String query_rewritten = SparqlRewriter.rewriteQuery(query);
548 			result = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(query_rewritten, format);
549 		}
550 		else {
551 			result = getSelectConstructAskResponseClassic(query, format);
552 		}
553 		
554 		if (format.equals("text/html")){
555 			responseBuilder.entity(getHTMLResult(result, query));
556 		} else {
557 			responseBuilder.entity(result);
558 		}
559 		responseBuilder.type(format);
560 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
561 		return responseBuilder.build();
562 	}

566 	 * @param query

568 	 * @param result
569 	 */
570 	private String getHTMLResult(final String result, String query) {
571 		MustacheFactory mf = new DefaultMustacheFactory();

589

591 		while (m.find()) {
592 			String graphName = m.group("graph");
593 			String type = m.group("type");
594 			String revisionNumber = m.group("revision").toLowerCase();
595 			String newGraphName;
596 
597 			// if no revision number is declared use the MASTER as default
598 			if (revisionNumber == null) {
599 				revisionNumber = "master";
600 			}
601 			if (revisionNumber.equalsIgnoreCase("master")) {
602 				// Respond with MASTER revision - nothing to be done - MASTER revisions are already created in the named graphs
603 				newGraphName = graphName;
604 			} else {
605 				if (RevisionManagement.isBranch(graphName, revisionNumber)) {
606 					newGraphName = RevisionManagement.getReferenceGraph(graphName, revisionNumber);
607 				} else {
608 					// Respond with specified revision, therefore the revision must be generated - saved in graph <RM-TEMP-graphName>
609 					newGraphName = graphName + "-temp";
610 					RevisionManagement.generateFullGraphOfRevision(graphName, revisionNumber, newGraphName);
611 				}
612 			}
613 
614 			queryM = m.replaceFirst(type + " <" + newGraphName + ">");
615 			m = patternSelectFromPart.matcher(queryM);
616 			
617 		}
618 		String response = TripleStoreInterfaceSingleton.get().executeSelectConstructAskQuery(queryM, format);
619 		return response;
620 	}

624 	 * Produce the response for a INSERT or DELETE SPARQL query.
625 	 * 
626 	 * @param query
627 	 *            the SPARQL query containing the update request
628 	 * @param user
629 	 * 			  user committing the update
630 	 * @param commitMessage
631 	 * 			  message describing the commit
632 	 * @param format
633 	 *            the result format
634 	 * @return the response with HTTP header for every graph (revision number
635 	 *         and MASTER revision number)
636 	 * @throws InternalErrorException 
637 	 */
638 	private Response getUpdateResponse(final String query, final String user, final String commitMessage,
639 			final String format) throws InternalErrorException {
640 		logger.info("Update detected");
641 		
642 		// write to add and delete sets
643 		// (replace graph names in query)
644 		String queryM = query;
645 		Matcher m = patternUpdateRevision.matcher(queryM);
646 		while (m.find()) {
647 			String graphName = m.group("graph");
648 			String revisionName = m.group("revision").toLowerCase(); 	// can contain revision
649 																		// numbers or reference
650 																		// names
651 			String action = m.group("action");
652 			String data = m.group("data");
653 			if (data == null)
654 				data = "";
655 			
656 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
657 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
658 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
659 			if (!RevisionManagement.isBranch(graphName, revisionName)) {
660 				throw new InternalErrorException("Revision is not referenced by a branch");
661 			}
662 			if (action.equalsIgnoreCase("INSERT")) {
663 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, addSetGraphUri));
664 			} else if (action.equalsIgnoreCase("DELETE")) {
665 				queryM = m.replaceFirst(String.format("INSERT %s { GRAPH <%s>", data, removeSetGraphUri));
666 			} else if (action.equalsIgnoreCase("WHERE")) {
667 				// TODO: replace generateFullGraphOfRevision with SPARQL JOIN
668 				String tempGraphName = graphName + "-temp";
669 				RevisionManagement.generateFullGraphOfRevision(graphName, revisionName, tempGraphName);
670 				queryM = m.replaceFirst(String.format("WHERE { GRAPH <%s>", tempGraphName));
671 			}
672 			m = patternUpdateRevision.matcher(queryM);
673 		}
674 		
675 		// Remove empty insert clauses which otherwise will lead to errors
676 		m= patternEmptyGraphPattern.matcher(queryM);
677 		queryM = m.replaceAll("");
678 
679 		TripleStoreInterfaceSingleton.get().executeUpdateQuery(queryM);
680 
681 		queryM = query;
682 		m = patternGraphWithRevision.matcher(queryM);
683 		while (m.find()) {
684 			String graphName = m.group("graph");
685 			String revisionName = m.group("revision").toLowerCase();	// can contain revision
686 																		// numbers or reference
687 																		// names
688 			// General variables
689 			String newRevisionNumber = RevisionManagement.getNextRevisionNumber(graphName, revisionName);
690 			String referenceFullGraph = RevisionManagement.getReferenceGraph(graphName, revisionName);
691 			String addSetGraphUri = graphName + "-delta-added-" + newRevisionNumber;
692 			String removeSetGraphUri = graphName + "-delta-removed-" + newRevisionNumber;
693 
694 			// remove doubled data
695 			// (already existing triples in add set; not existing triples in
696 			// delete set)
697 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
698 							"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", addSetGraphUri,
699 							referenceFullGraph));
700 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
701 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } MINUS { GRAPH <%s> { ?s ?p ?o. } } }",
702 					removeSetGraphUri, removeSetGraphUri, referenceFullGraph));
703 
704 			// merge change sets into reference graph
705 			// (copy add set to reference graph; remove delete set from reference graph)
706 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
707 						"INSERT { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }",
708 						referenceFullGraph,	addSetGraphUri));
709 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(String.format(
710 					"DELETE { GRAPH <%s> { ?s ?p ?o. } } WHERE { GRAPH <%s> { ?s ?p ?o. } }", 
711 					referenceFullGraph,	removeSetGraphUri));
712 
713 			// add meta information to R43ples
714 			ArrayList<String> usedRevisionNumber = new ArrayList<String>();
715 			usedRevisionNumber.add(revisionName);
716 			RevisionManagement.addMetaInformationForNewRevision(graphName, user, commitMessage, usedRevisionNumber,
717 					newRevisionNumber, addSetGraphUri, removeSetGraphUri);
718 			
719 			
720 			queryM = m.replaceAll(String.format("GRAPH <%s> ", graphName));
721 			m = patternGraphWithRevision.matcher(queryM);
722 		}
723 		
724 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
725 		
726 		String result = "Query successfully executed";
727 		if (format.equals("text/html")){
728 			responseBuilder.entity(getHTMLResult(result, query));
729 		} else {
730 			responseBuilder.entity(result);
731 		}
732 		responseBuilder.type(format);
733 		responseBuilder.header("r43ples-revisiongraph", RevisionManagement.getResponseHeaderFromQuery(query));		
734 		return responseBuilder.build();
735 	}

867 	/** 
868 	 * Creates a merge between the specified branches.
869 	 * 
870 	 * Using command: MERGE GRAPH <graphURI> BRANCH "branchNameA" INTO "branchNameB"
871 	 * 
872 	 * @param sparqlQuery the SPARQL query
873 	 * @param format the result format
874 	 * @throws InternalErrorException 
875 	 */
876 	private Response getMergeResponse(final String sparqlQuery, final String user, final String commitMessage, final String format) throws InternalErrorException {
877 		ResponseBuilder responseBuilder = Response.created(URI.create(""));
878 		logger.info("Merge creation detected");
879 
880 		// Add R43ples information
881 		Matcher m = patternMergeQuery.matcher(sparqlQuery);
882 		
883 		boolean foundEntry = false;
884 		while (m.find()) {
885 			foundEntry = true;
886 			String newRevisionNumber = null;
887 			
888 			String action = m.group("action");
889 			String graphName = m.group("graph");
890 			String sdd = m.group("sdd");
891 			String sddURI = m.group("sddURI");
892 			String branchNameA = m.group("branchNameA").toLowerCase();
893 			String branchNameB = m.group("branchNameB").toLowerCase();
894 			String with = m.group("with");
895 			String triples = m.group("triples");
896 			
897 			String revisionUriA = RevisionManagement.getRevisionUri(graphName, branchNameA);
898 			String revisionUriB = RevisionManagement.getRevisionUri(graphName, branchNameB);
899 			
900 			logger.debug("action: " + action);
901 			logger.debug("graph: " + graphName);
902 			logger.debug("sdd: " + sdd);
903 			logger.debug("sddURI: " + sddURI);
904 			logger.debug("branchNameA: " + branchNameA);
905 			logger.debug("branchNameB: " + branchNameB);
906 			logger.debug("with: " + with);
907 			logger.debug("triples: " + triples);
908 			
909 			if (!RevisionManagement.checkGraphExistence(graphName)){
910 				logger.error("Graph <"+graphName+"> does not exist.");
911 				throw new InternalErrorException("Graph <"+graphName+"> does not exist.");
912 			}
913 				
914 			
915 			// Check if A and B are different revisions
916 			if (RevisionManagement.getRevisionNumber(graphName, branchNameA).equals(RevisionManagement.getRevisionNumber(graphName, branchNameB))) {
917 				// Branches are equal - throw error
918 				throw new InternalErrorException("Specified branches are equal: " + sparqlQuery);
919 			}
920 			
921 			// Check if both are terminal nodes
922 			if (!(RevisionManagement.isBranch(graphName, branchNameA) && RevisionManagement.isBranch(graphName, branchNameB))) {
923 				throw new InternalErrorException("Non terminal nodes were used: " + sparqlQuery);
924 			}
925 
926 			
927 			// Differ between MERGE query with specified SDD and without SDD			
928 			String usedSDDURI = null;
929 			if (sdd != null) {
930 				// Specified SDD
931 				usedSDDURI = sddURI;
932 			} else {
933 				// Default SDD
934 				// Query the referenced SDD
935 				String querySDD = String.format(
936 						  "PREFIX sddo: <http://eatld.et.tu-dresden.de/sddo#> %n"
937 						+ "PREFIX rmo: <http://eatld.et.tu-dresden.de/rmo#> %n"
938 						+ "SELECT ?defaultSDD %n"
939 						+ "WHERE { GRAPH <%s> {	%n"
940 						+ "	<%s> a rmo:Graph ;%n"
941 						+ "		sddo:hasDefaultSDD ?defaultSDD . %n"
942 						+ "} }", Config.revision_graph, graphName);
943 				
944 				ResultSet resultSetSDD = TripleStoreInterfaceSingleton.get().executeSelectQuery(querySDD);
945 				if (resultSetSDD.hasNext()) {
946 					QuerySolution qs = resultSetSDD.next();
947 					usedSDDURI = qs.getResource("?defaultSDD").toString();
948 				} else {
949 					throw new InternalErrorException("Error in revision graph! Selected graph <" + graphName + "> has no default SDD referenced.");
950 				}
951 			}
952 
953 			// Get the common revision with shortest path
954 			String commonRevision = MergeManagement.getCommonRevisionWithShortestPath(revisionUriA, revisionUriB);
955 			
956 			// Create the revision progress for A and B
957 			String graphNameA = graphName + "-RM-REVISION-PROGRESS-A";
958 			String graphNameB = graphName + "-RM-REVISION-PROGRESS-B";
959 			String graphNameDiff = graphName + "-RM-DIFFERENCE-MODEL";
960 			String uriA = "http://eatld.et.tu-dresden.de/branch-A";
961 			String uriB = "http://eatld.et.tu-dresden.de/branch-B";
962 			
963 			MergeManagement.createRevisionProgresses(MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriA), graphNameA, uriA, MergeManagement.getPathBetweenStartAndTargetRevision(commonRevision, revisionUriB), graphNameB, uriB);
964 			
965 			// Create difference model
966 			MergeManagement.createDifferenceTripleModel(graphName,  graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI);
967 			
968 			// Differ between the different merge queries
969 			if ((action != null) && (action.equalsIgnoreCase("AUTO")) && (with == null) && (triples == null)) {
970 				logger.info("AUTO MERGE query detected");
971 				// Create the merged revision
972 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.AUTO, "");
973 			} else if ((action != null) && (action.equalsIgnoreCase("MANUAL")) && (with != null) && (triples != null)) {
974 				logger.info("MANUAL MERGE query detected");
975 				// Create the merged revision
976 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.MANUAL, triples);
977 			} else if ((action == null) && (with != null) && (triples != null)) {
978 				logger.info("MERGE WITH query detected");
979 				// Create the merged revision
980 				newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.WITH, triples);
981 			} else if ((action == null) && (with == null) && (triples == null)) {
982 				logger.info("MERGE query detected");
983 				// Check if difference model contains conflicts
984 				String queryASK = String.format(
985 						  "ASK { %n"
986 						+ "	GRAPH <%s> { %n"
987 						+ " 	?ref <http://eatld.et.tu-dresden.de/sddo#isConflicting> \"true\"^^<http://www.w3.org/2001/XMLSchema#boolean> . %n"
988 						+ "	} %n"
989 						+ "}", graphNameDiff);
990 				if (TripleStoreInterfaceSingleton.get().executeAskQuery(queryASK)) {
991 					// Difference model contains conflicts
992 					// Return the conflict model to the client
993 					responseBuilder = Response.status(Response.Status.CONFLICT);
994 					responseBuilder.entity(RevisionManagement.getContentOfGraphByConstruct(graphNameDiff, format));
995 				} else {
996 					// Difference model contains no conflicts
997 					// Create the merged revision
998 					newRevisionNumber = MergeManagement.createMergedRevision(graphName, branchNameA, branchNameB, user, commitMessage, graphNameDiff, graphNameA, uriA, graphNameB, uriB, usedSDDURI, MergeQueryTypeEnum.COMMON, "");
999 				}
1000 			} else {
1001 				throw new InternalErrorException("This is not a valid MERGE query: " + sparqlQuery);
1002 			}
1003 			
1004 			String graphNameHeader;
1005 			try {
1006 				graphNameHeader = URLEncoder.encode(graphName, "UTF-8");
1007 			} catch (UnsupportedEncodingException e) {
1008 				e.printStackTrace();
1009 				graphNameHeader = graphName;
1010 			}
1011 			
1012 			
1013 			// Return the revision number which were used (convert tag or branch identifier to revision number)
1014 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-A", RevisionManagement.getRevisionNumber(graphName, branchNameA));
1015 			responseBuilder.header(graphNameHeader + "-revision-number-of-branch-B", RevisionManagement.getRevisionNumber(graphName, branchNameB));			
1016 			
1017 			if (newRevisionNumber != null) {
1018 				// Respond with next revision number
1019 		    	responseBuilder.header(graphNameHeader + "-revision-number", newRevisionNumber);
1020 				responseBuilder.header(graphNameHeader + "-revision-number-of-MASTER", RevisionManagement.getMasterRevisionNumber(graphName));
1021 				logger.debug("Respond with new revision number " + newRevisionNumber + ".");
1022 			}
1023 		}
1024 		if (!foundEntry)
1025 			throw new InternalErrorException("Error in query: " + sparqlQuery);
1026 		
1027 		return responseBuilder.build();	
1028 	}
