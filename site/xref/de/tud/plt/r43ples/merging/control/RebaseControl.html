64  	/**for each revision in branchA , create a patch */
65  	public void createPatchGroupOfBranch(String revisionGraph, String basisRevisionUri, LinkedList<String> revisionList) {
66  		
67  		LinkedHashMap<String, Patch> patchMap = new LinkedHashMap<String, Patch>();
68  		
69  		Iterator<String> rIter  = revisionList.iterator();
70  		
71  		while(rIter.hasNext()) {
72  			String revisionUri = rIter.next();
73  			String commitUri = StrategyManagement.getCommitUri(revisionGraph, revisionUri);
74  			
75  			String addSet = StrategyManagement.getaddSetUri(revisionGraph, revisionUri);
76  			String deleteSet = StrategyManagement.getdeleteSetUri(revisionGraph, revisionUri);
77  			
78  			String patchNumber = StrategyManagement.getRevisionNumber(revisionGraph, revisionUri);
79  			String patchUser = StrategyManagement.getPatchUserUri(revisionGraph, commitUri);
80  			String patchMessage = StrategyManagement.getPatchMessage(revisionGraph, commitUri);
81  			
82  			patchMap.put(patchNumber, new Patch(patchNumber, patchUser, patchMessage, addSet, deleteSet));			
83  		}
84  		
85  		String basisRevisionNumber = StrategyManagement.getRevisionNumber(revisionGraph, basisRevisionUri);
86  		
87  		patchGroup.setBasisRevisionNumber(basisRevisionNumber);
88  		patchGroup.setPatchMap(patchMap);
89  		
90  		logger.info("patchGroup initial successful!" + patchGroup.getPatchMap().size());
91  	}

93  	/**
94  	 * force rebase begin, for each patch in patch group will a new revision created 
95  	 * @throws InternalErrorException 
96  	 * */
97  	public String forceRebaseProcess( String graphName ) throws InternalErrorException{
98  		
99  		logger.info("patchGroup 1:" + patchGroup.getBasisRevisionNumber());
100 		logger.info("patchGroup 2:" + patchGroup.getPatchMap().size());
101 
102 		LinkedHashMap<String, Patch> patchMap = patchGroup.getPatchMap();
103 		String basisRevisionNumber = patchGroup.getBasisRevisionNumber();
104 				
105 		Iterator<Entry<String, Patch>> pIter = patchMap.entrySet().iterator();
106 		
107 		while(pIter.hasNext()) {
108 			Entry<String, Patch> pEntry = pIter.next();
109 			Patch patch = pEntry.getValue();
110 		
111 			String newRevisionNumber = RevisionManagement.createNewRevisionWithPatch(
112 					graphName, patch.getAddedSetUri(), patch.getRemovedSetUri(),
113 					patch.getPatchUser(), patch.getPatchMessage(), basisRevisionNumber);
114 			
115 			basisRevisionNumber = newRevisionNumber;
116 		}
117 		return basisRevisionNumber;	
118 	}

141 	/**read the difference model and set the automatic resolution state for the triple and create */
142 	public ArrayList<String> getAutomaticAddedTriplesAndRemovedTriples() {
143 		ArrayList<String> list = new ArrayList<String>();
144 		
145 		String triplesToInsert = "";
146 		String triplesToDelete = "";
147 		
148 		// Iterate over all difference groups
149 		Iterator<String> iteDifferenceGroups = differenceModel.getDifferenceGroups().keySet().iterator();
150 		while (iteDifferenceGroups.hasNext()) {
151 			String differenceGroupKey = iteDifferenceGroups.next();
152 			DifferenceGroup differenceGroup = differenceModel.getDifferenceGroups().get(differenceGroupKey);
153 			
154 			// Get the triple state to use
155 			SDDTripleStateEnum tripleState;
156 			
157 			// Use the automatic resolution state
158 			tripleState = differenceGroup.getAutomaticResolutionState();
159 		
160 			
161 			Iterator<String> iteDifferences = differenceGroup.getDifferences().keySet().iterator();
162 			while (iteDifferences.hasNext()) {
163 				String differenceKey = iteDifferences.next();
164 				Difference difference = differenceGroup.getDifferences().get(differenceKey);
165 								
166 				// Get the triple
167 				String triple = ProcessManagement.tripleToString( difference.getTriple());					
168 				
169 				// Add the triple to the corresponding string
170 				if (tripleState.equals(SDDTripleStateEnum.ADDED) || tripleState.equals(SDDTripleStateEnum.ORIGINAL)) {
171 					triplesToInsert += triple + "%n";				
172 				} else if (tripleState.equals(SDDTripleStateEnum.DELETED) || tripleState.equals(SDDTripleStateEnum.NOTINCLUDED)) {
173 					triplesToDelete += triple + "%n";
174 				} else {
175 					// Error occurred - state was not changed
176 					logger.error("Triple state was used which has no internal representation.");
177 				}
178 			}
179 		}
180 		
181 		// Add the string to the result list
182 		list.add(String.format(triplesToInsert));
183 		list.add(String.format(triplesToDelete));
184 		
185 		logger.info("automatic insert list: "+ list.get(0).toString());
186 		logger.info("automatic delete list: "+ list.get(1).toString());
187 
188 		
189 		return list;
190 	}

193 	/**read the difference model and set the automatic resolution state for the triple and create */
194 	public ArrayList<String> getManualAddedTriplesAndRemovedTriples() {
195 		// The result list
196 		ArrayList<String> list = new ArrayList<String>();
197 		
198 		String triplesToInsert = "";
199 		String triplesToDelete = "";
200 		
201 		// Iterate over all difference groups
202 		Iterator<String> iteDifferenceGroups = differenceModel.getDifferenceGroups().keySet().iterator();
203 		while (iteDifferenceGroups.hasNext()) {
204 			String differenceGroupKey = iteDifferenceGroups.next();
205 			DifferenceGroup differenceGroup = differenceModel.getDifferenceGroups().get(differenceGroupKey);
206 			
207 			Iterator<String> iteDifferences = differenceGroup.getDifferences().keySet().iterator();
208 			while (iteDifferences.hasNext()) {
209 				String differenceKey = iteDifferences.next();
210 				Difference difference = differenceGroup.getDifferences().get(differenceKey);
211 				
212 				// Get the triple state to use
213 				SDDTripleStateEnum tripleState;
214 				if (difference.getResolutionState().equals(ResolutionStateEnum.RESOLVED)) {
215 					// Use the approved triple state
216 					tripleState = difference.getTripleResolutionState();					
217 				} else {
218 					// Use the automatic resolution state
219 					tripleState = differenceGroup.getAutomaticResolutionState();
220 				}
221 				
222 				// Get the triple
223 				String triple = ProcessManagement.tripleToString( difference.getTriple());					
224 				
225 				// Add the triple to the corresponding string
226 				if (tripleState.equals(SDDTripleStateEnum.ADDED) || tripleState.equals(SDDTripleStateEnum.ORIGINAL)) {
227 					triplesToInsert += triple + "%n";				
228 				} else if (tripleState.equals(SDDTripleStateEnum.DELETED) || tripleState.equals(SDDTripleStateEnum.NOTINCLUDED)) {
229 					triplesToDelete += triple + "%n";
230 				} else {
231 					// Error occurred - state was not changed
232 					logger.error("Triple state was used which has no internal representation.");
233 				}
234 			}
235 		}
236 		
237 		// Add the string to the result list
238 		list.add(String.format(triplesToInsert));
239 		list.add(String.format(triplesToDelete));
240 		
241 		return list;
242 	}
