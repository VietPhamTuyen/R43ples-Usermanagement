19  	/**
20  	 * get the report page of fast forward query
21  	 * @param graphName
22  	 * */
23  
24  	public String getFastForwardReportView(String graphName) {
25  		Map<String, Object> scope = new HashMap<String, Object>();
26  		MustacheFactory mf = new DefaultMustacheFactory();
27  		Mustache mustache = mf.compile("templates/merge/mergingResultView.mustache");
28  		StringWriter sw = new StringWriter();		
29  		scope.put("graphName", graphName);
30  		scope.put("commit", commitModel);
31  		scope.put("merging_active", true);
32  		
33  		mustache.execute(sw, scope);	
34  		return sw.toString();	
35  	}
36  	
37  	/**check the condition for fast forward strategy
38  	 * @param graphName : name of named graph
39  	 * @param branch1 : name of branch1
40  	 * @param branch2 : name of branch2
41  	 * @throws InternalErrorException */
42  	
43  	public static boolean fastForwardCheck(String graphName, String branch1 , String branch2) throws InternalErrorException {
44  		if(branch1.equals(branch2)) {
45  			return false;
46  		}
47  		
48  		//get last revision of each branch
49  		String revisionUriA = RevisionManagement.getRevisionUri(graphName, branch1);
50  		String revisionUriB = RevisionManagement.getRevisionUri(graphName, branch2);
51  		
52  		return StrategyManagement.isFastForward(revisionUriA, revisionUriB);
53  	}
54  	
55

57  		commitModel = new CommitModel(graphName, sddName, user, message, branch1, branch2, strategy, type);
58  	}
