534 	 * @param id: to approved Triple id 
535 	 * @param isChecked : status of approved Triple*/
536 	
537 	public void approveToDifferenceModel(String id, String isChecked){
538 		
539 		// Count for Conflict;
540 		int count = 0;
541 		Triple checkedTriple = tableModel.getManuellTriple().get(id);
542 		SDDTripleStateEnum tripleState;
543 		if (isChecked.equals("1")){
544 			tripleState = SDDTripleStateEnum.ADDED;
545 		}else {
546 			tripleState = SDDTripleStateEnum.DELETED;
547 		}
548 		
549 		Iterator<Entry<String, DifferenceGroup>> iterDM = differenceModel.getDifferenceGroups().entrySet().iterator();
550 		while(iterDM.hasNext()) {
