1   package de.tud.plt.r43ples.merging.management;
2   
3   import java.io.File;
4   import java.util.HashMap;
5   import java.util.Iterator;
6   
7   import org.apache.commons.configuration.ConfigurationException;
8   import org.apache.commons.configuration.PropertiesConfiguration;
9   
10  /**
11   * Reads and stores the configuration parameters.
12   * 
13   * @author Stephan Hensel
14   *
15   */
16  public class ClientConfig {
17  	
18  	/** The properties configuration. **/
19  	private static PropertiesConfiguration config;
20  	/** The R43ples SPARQL endpoint. **/
21  	public static String r43ples_sparql_endpoint;
22  	/** The R43ples JSON HTTP-GET interface to get all revised graphs. **/
23  	public static String r43ples_json_revisedgraphs;
24  	/** The R43ples revision graph. **/
25  	public static String r43ples_revision_graph;
26  	/** The R43ples SDD graph. **/
27  	public static String r43ples_sdd_graph;
28  	/** The client prefix mappings. (key: mapping, value: prefix) **/
29  	public static HashMap<String, String> prefixMappings = new HashMap<String, String>();
30  	
31  	
32  	/**
33  	 * Read the configuration information from local file.
34  	 * 
35  	 * @param configFilePath path to configuration file
36  	 * @throws ConfigurationException error about configuration 
37  	 */
38  	public static void readConfig(final String configFilePath) throws ConfigurationException{
39  		config = new PropertiesConfiguration(configFilePath);
40  		r43ples_sparql_endpoint = config.getString("r43ples.sparql.endpoint");
41  		r43ples_json_revisedgraphs = config.getString("r43ples.json.revisedgraphs");
42  		r43ples_revision_graph = config.getString("r43ples.revision.graph");
43  		r43ples_sdd_graph = config.getString("r43ples.sdd.graph");
44  		String[] prefixMappingsArray = config.getStringArray("client.prefixmapping");
45  		// Generate hash map of prefix mappings
46  		prefixMappings.clear();
47  		for (int i = 0; i < prefixMappingsArray.length; i++) {
48  			String mapping = prefixMappingsArray[i];
49  			String[] parts = mapping.split("\\|");
50  			prefixMappings.put(parts[1], parts[0]);
51  		}
52  	}
53  	
54  	
55  	/**
56  	 * Write the configuration information to local file.
57  	 * 
58  	 * @param configFilePath path to configuration file
59  	 * @throws ConfigurationException
60  	 */
61  	public static void writeConfig(final String configFilePath) throws ConfigurationException {
62  		config.setProperty("r43ples.sparql.endpoint", r43ples_sparql_endpoint);
63  		config.setProperty("r43ples.json.revisedgraphs", r43ples_json_revisedgraphs);
64  		config.setProperty("r43ples.revision.graph", r43ples_revision_graph);
65  		config.setProperty("r43ples.sdd.graph", r43ples_sdd_graph);
66  		// Generate array of prefix mappings
67  		String[] prefixMappingsArray = new String[prefixMappings.size()];
68  		int index = 0;
69  		Iterator<String> itePrefixMappingsKeys = prefixMappings.keySet().iterator();
70  		while (itePrefixMappingsKeys.hasNext()) {
71  			String currentKey = itePrefixMappingsKeys.next();
72  			String currentValue = prefixMappings.get(currentKey);
73  			prefixMappingsArray[index] = currentValue + "|" + currentKey;
74  			index++;			
75  		}
76  		config.clearProperty("client.prefixmapping");
77  		config.setProperty("client.prefixmapping", prefixMappingsArray);
78  		config.save(new File(configFilePath));
79  	}
80  
81  }
