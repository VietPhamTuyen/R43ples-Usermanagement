26  	/**initial report result and create report list
27  	 * @param difference model
28  	 * */
29  	public static ReportResult initialReportResult(DifferenceModel differenceModel) {
30  		ReportResult reportResult = new ReportResult();
31  		
32  		
33  		// Count for Conflict;
34  		int count = 0;
35  		
36  		Iterator<Entry<String, DifferenceGroup>> iterDM = differenceModel.getDifferenceGroups().entrySet().iterator();
37  		while(iterDM.hasNext()) {
38  			Entry<String, DifferenceGroup> entryDG = (Entry<String, DifferenceGroup>) iterDM.next();
39  			DifferenceGroup differ = (DifferenceGroup) entryDG.getValue();
40  			
41  			Iterator<Entry<String, Difference>> iterDIF = differ.getDifferences().entrySet().iterator();
42  			while(iterDIF.hasNext()){
43  				Entry<String, Difference> entryDF = iterDIF.next();
44  								
45  				Difference difference = entryDF.getValue();
46  				
47  				if(difference.getResolutionState().equals(ResolutionStateEnum.CONFLICT)){
48  					count ++;
49  				}
50  				
51  			}
52  		}
53  		//initial number of Triple with Conflict	
54  		reportResult.setConflictsNotApproved(count);
55  		logger.info("initial Result Report: "+ reportResult.getConflictsNotApproved()+"--"+reportResult.getDifferencesResolutionChanged());
56  		
57  		return reportResult;
58  	}

61  	/**read neue difference model and create report table row list
62  	 * @param difference model
63  	 * @throws ConfigurationException 
64  	 * */
65  	public static List<ReportTableRow> createReportTableRowList(DifferenceModel differenceModel) throws ConfigurationException{
66  		
67  		//new a List for ReportTableRow
68  		List<ReportTableRow> reportTableRowList = new ArrayList<ReportTableRow>();
69   		
70  		//get difference group
71  		Iterator<Entry<String, DifferenceGroup>> iterDM = differenceModel.getDifferenceGroups().entrySet().iterator();
72  		while(iterDM.hasNext()){
73  			Entry<String, DifferenceGroup> entryDG = (Entry<String, DifferenceGroup>) iterDM.next();
74  			DifferenceGroup differ = (DifferenceGroup) entryDG.getValue();
75  			boolean isconflicting = differ.isConflicting();
76  			SDDTripleStateEnum stateA = differ.getTripleStateA();
77  			SDDTripleStateEnum stateB = differ.getTripleStateB();
78  			
79  			SDDTripleStateEnum automaticResolutionState = differ.getAutomaticResolutionState();
80  			
81  			String conflicting = (isconflicting ) ? "1" : "0";
82  			//get difference 
83  			Iterator<Entry<String, Difference>> iterDIF = differ.getDifferences().entrySet().iterator();
84  			while(iterDIF.hasNext()){
85  				
86  				Entry<String, Difference> entryDF = iterDIF.next();
87  				
88  				Difference difference = entryDF.getValue();
89  				//get triple
90  				Triple triple = difference.getTriple();
91  				
92  				ResolutionStateEnum approvedState = difference.getResolutionState();
93  				
94  				SDDTripleStateEnum tripleResolutionState = difference.getTripleResolutionState();
95  				
96  				String subject = ProcessManagement.convertTripleStringToPrefixTripleString(ProcessManagement.getSubject(triple));
97  				String predicate = ProcessManagement.convertTripleStringToPrefixTripleString(ProcessManagement.getPredicate(triple));
98  				String object = ProcessManagement.convertTripleStringToPrefixTripleString(ProcessManagement.getObject(triple));
99  				//get revision number
100 				String revisionA = difference.getReferencedRevisionLabelA();
101 				String revisionB = difference.getReferencedRevisionLabelB();
102 				
103 				//read each reportTableRowList
104 				
105 				reportTableRowList.add(new ReportTableRow(subject, predicate, object, stateA.toString(), stateB.toString(), revisionA, revisionB, 
106 						conflicting, automaticResolutionState.toString(), tripleResolutionState.toString(), approvedState.toString()));
107 																	
108 			}			
109 			
110 		}
111 		// test report table row list
112 		Iterator<ReportTableRow> idt = reportTableRowList.iterator();
113 		while(idt.hasNext()){
114 			ReportTableRow tr = idt.next();
115 			logger.info("TableModel ID Test:" + tr.getRevisionA() + tr.getResolutionState() + tr.getApproved());
116 		}
117 		
118 		
119 		logger.info("TableModel successful created.");
120 		
121 		return reportTableRowList;
122 			
123 	}
