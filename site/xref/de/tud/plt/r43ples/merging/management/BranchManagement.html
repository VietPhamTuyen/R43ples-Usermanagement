5   import org.apache.log4j.Logger;

10  import de.tud.plt.r43ples.management.RevisionManagement;

13  
14  public class BranchManagement {
15  	private static Logger logger = Logger.getLogger(BranchManagement.class);
16  	
17  	/**
18  	 * get the names of all branches of a named graph
19  	 * @param graphName name of graph
20  	 * @return list of the name of branch*/
21  	public static ArrayList<String> getAllBranchNamesOfGraph(String graphName) {
22  		logger.info("Get all branch names of graph "+ graphName);
23  		ArrayList<String> list = new ArrayList<String>();	
24  		if (graphName != null) {
25  			String revisionGraph = RevisionManagement.getRevisionGraph(graphName);
26  			String sparqlQuery = RevisionManagement.prefixes
27  					+ String.format(
28  					  "SELECT DISTINCT ?label %n"
29  					+ "FROM <%s> %n"
30  					+ "WHERE { %n"
31  					+ "	?branch a rmo:Branch ;"
32  					+ "		rdfs:label ?label . "
33  					+ "} %n"
34  					+ "ORDER BY ?label", revisionGraph);
35  			
36  			ResultSet results = TripleStoreInterfaceSingleton.get().executeSelectQuery(sparqlQuery);
37  			while (results.hasNext()) {
38  				QuerySolution qs = results.next();
39  				list.add(qs.getLiteral("label").toString());
40  			}		
41  		}
42  		logger.debug("All branches: " +list);
43  		return list;
44  	}
45  }
