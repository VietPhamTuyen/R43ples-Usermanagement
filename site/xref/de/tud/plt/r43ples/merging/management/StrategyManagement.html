37  	 * @param last revision uri of branch A
38  	 * @param last revision uri of branch B
39  	 * */
40  	public static boolean isFastForward(String revisionBranchA , String revisionBranchB){
41  
42  		
43  		String query = prefixes
44  				+ String.format("ASK { GRAPH <%s> { "
45  						+ "<%s> prov:wasDerivedFrom+ <%s> ."
46  						+ " }} ",
47  						Config.revision_graph, revisionBranchA, revisionBranchB);
48  			
49  		return TripleStoreInterfaceSingleton.get().executeAskQuery(query);
50  			
51  	}

78  	 * @param name of branch B
79  	 * @param uri of branch B
80  	 * @param uri of branch A
81  	 *  */
82  	public static void moveBranchReference(String branchNameB, String revisionUriB, String revisionUriA){
83  		// delete alte reference
84  		String query = prefixes + String.format("DELETE DATA { GRAPH <%s> { <%s> rmo:references <%s>. } };%n",
85  				Config.revision_graph, branchNameB, revisionUriB);
86  		// added new reference
87  		query += String.format("INSERT DATA { GRAPH <%s> { <%s> rmo:references <%s>. } } ;%n", Config.revision_graph,
88  				branchNameB, revisionUriA);
89  		
90  		logger.info("move info" + query);
91  		TripleStoreInterfaceSingleton.get().executeUpdateQuery(query);
92  				
93  	}

97  	 * @param uri of branch B
98  	 * @param uri of last revision of branch B
99  	 * @param uri of last revision of branch A
100 	 * */
101 	public static void updateRevisionOfBranch(String branchUriB, String revisionUriB, String revisionUriA ){
102 		LinkedList<String> revisionList =  MergeManagement.getPathBetweenStartAndTargetRevision(revisionUriB, revisionUriA);
103 		
104 		logger.info("revisionlist size: "+ revisionList.size());
105 		Iterator<String> riter = revisionList.iterator();
106 		while(riter.hasNext()) {
107 			String revision = riter.next();
108 			String query = prefixes + String.format("INSERT DATA { GRAPH <%s> { <%s> rmo:revisionOfBranch <%s>. } };%n",
109 					Config.revision_graph, revision, branchUriB);
110 			
111 			logger.info("revisionlist info" + revision);
112 
113 			logger.info("updated info" + query);
114 			TripleStoreInterfaceSingleton.get().executeUpdateQuery(query);			
115 		}
116

165 	/**create rebase query
166 	 * @param graphName
167 	 * @param sdd model name
168 	 * @param client name 
169 	 * @param messsage of client
170 	 * @param name of branch A
171 	 * @param name of branch B
172 	 * @param type of rebase
173 	 * @param triple set*/
174 	public static String createRebaseQuery(String graphName, String sdd, String user, String commitMessage, String branchNameA, String branchNameB, RebaseQueryTypeEnum type, String triples){
175 		logger.info("Execute rebase query of type " + type.toString());
176 		String query = "";
177 		
178 		String queryTemplateCommon = 
179 				  "USER \"%s\" %n"
180 				+ "MESSAGE \"%s\" %n"
181 				+ "REBASE GRAPH <%s> SDD <%s> BRANCH \"%s\" INTO \"%s\"";
182 				
183 		String queryTemplateAuto = 
184 				  "USER \"%s\" %n"
185 				+ "MESSAGE \"%s\" %n"
186 				+ "REBASE AUTO GRAPH <%s> SDD <%s> BRANCH \"%s\" INTO \"%s\"";
187 		
188 		String queryTemplateForce = 
189 				  "USER \"%s\" %n"
190 				+ "MESSAGE \"%s\" %n"
191 				+ "REBASE FORCE GRAPH <%s> SDD <%s> BRANCH \"%s\" INTO \"%s\"";
192 		
193 		String queryTemplateManual = 
194 				  "USER \"%s\" %n"
195 				+ "MESSAGE \"%s\" %n"
196 				+ "REBASE MANUAL GRAPH <%s> SDD <%s> BRANCH \"%s\" INTO \"%s\" WITH { %n"
197 				+ "	%s"
198 				+ "}";
199 		
200 		String queryTemplateWith = 
201 				  "USER \"%s\" %n"
202 				+ "MESSAGE \"%s\" %n"
203 				+ "REBASE GRAPH <%s> SDD <%s> BRANCH \"%s\" INTO \"%s\" WITH { %n"
204 				+ "	%s"
205 				+ "}";
206 		
207 		if (type.equals(RebaseQueryTypeEnum.COMMON)) {
208 			query = String.format(queryTemplateCommon, user, commitMessage, graphName, sdd, branchNameA, branchNameB);
209 		} else if (type.equals(RebaseQueryTypeEnum.AUTO)) {
210 			query = String.format(queryTemplateAuto, user, commitMessage, graphName, sdd, branchNameA, branchNameB);
211 		} else if (type.equals(RebaseQueryTypeEnum.FORCE)) {
212 			query = String.format(queryTemplateForce, user, commitMessage, graphName, sdd, branchNameA, branchNameB);
213 		} else if (type.equals(RebaseQueryTypeEnum.MANUAL)) {
214 			query = String.format(queryTemplateManual, user, commitMessage, graphName, sdd, branchNameA, branchNameB, triples);
215 		} else if (type.equals(RebaseQueryTypeEnum.WITH)) {
216 			query = String.format(queryTemplateWith, user, commitMessage, graphName, sdd, branchNameA, branchNameB, triples);
217 		}
218 		
219 		return query;
220 	}

222 	/**get commitUri of the revisionUri
223 	 * @param uri of the revision*/
224 	public static String getCommitUri(String revisionUri){
225 		
226 		String query = String.format(
227 				  "PREFIX prov: <http://www.w3.org/ns/prov#> %n"
228 				+ "SELECT DISTINCT ?commit  %n"
229 				+ "WHERE { %n"
230 				+ "	GRAPH <%s> { %n"
231 				+ "		?commit prov:generated <%s>."
232 				+ " }"
233 				+ "}", Config.revision_graph, revisionUri);
234 		
235 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
236 		
237 		if (resultSet.hasNext()) {
238 			QuerySolution qs = resultSet.next();
239 			return qs.getResource("?commit").toString();
240 		}
241 		
242 		logger.info("No commit could be found.");
243 		return null;
244 	}

286 	/** get the delta removed width versionUri
287 	 * @param uri of the added or removed triple set*/
288 	public static LinkedList<String> createAddedOrRemovedTripleSet(String addedOrRemovedDelta) {
289 		String query = prefixes + String.format(""
290 				+"SELECT DISTINCT ?s ?p ?o %n"
291 				+"WHERE{ GRAPH <%s> %n"
292 				+"		{ ?s ?p ?o . } %n"
293 				+"}", addedOrRemovedDelta);
294 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
295 		
296 		LinkedList<String> tripleList = new LinkedList<String>();
297 		while(resultSet.hasNext()) {
298 			QuerySolution qs = resultSet.next();
299 			String subject = "<" + qs.getResource("?s").toString() + ">";
300 			String predicate = "<" + qs.getResource("?p").toString() + ">";
301 			String object = "";
302 			
303 			if (qs.get("?o").isLiteral()) {
304 				object = "\"" + qs.getLiteral("?o").toString() + "\"";
305 			} else {
306 				object = "<" + qs.getResource("?o").toString() + ">";
307 			}
308 			
309 			String triple = subject + " " + predicate + " " + object;
310 			tripleList.add(triple);
311 			logger.info("patch--triple" + triple);
312 		}
313 		
314 		return tripleList;
315 	}

338 	 * @param uri of commit */
339 	public static String getPatchUserUri(String commitUri) {
340 		String query = prefixes + String.format(""
341 				+"SELECT DISTINCT ?user %n"
342 				+"WHERE{ GRAPH <%s> %n"
343 				+"   {<%s> prov:wasAssociatedWith ?user. } }%n",
344 				Config.revision_graph, commitUri);
345 		
346 		ResultSet resultSet = TripleStoreInterfaceSingleton.get().executeSelectQuery(query);
347 		
348 		if (resultSet.hasNext()) {
349 			QuerySolution qs = resultSet.next();
350 			String userUri = qs.getResource("?user").toString();		
351 			int count = userUri.split("/").length;
352 			return userUri.split("/")[count-1];
353 		}
354 		
355 		logger.info("No revision number could be found.");
356 		return null;
357 	}
